118template<unsigned int H, unsigned int W, typename T>
119MatBase<H, W, T>::MatBase(std::initializer_list<std::initializer_list<T>> v){
120 static_assert(std::is_arithmetic<T>::value, "Values must be numeric");
121 bool err = 0;
122 unsigned int size = 0;
123 try {
124 if(v.size() != H){
125 size = v.size();
126 throw "Matrix input height does not match templated height\nMatrix height: ";
127 }
128 for(auto vi = v.begin(); vi != v.end(); vi++){
129 for(auto vj = vi->begin(); vj != vi->end(); vj++){
130 if((unsigned int)(vj - vi->begin()) >= W){
131 err = 1;
132 size = vi->size();
133 throw "Matrix input width does not match templated width\nMatrix Width: ";
134 break;
135 }
136 (*this)[(int)(vi - v.begin())][(int)(vj - vi->begin())] = *vj;
137 }
138 }
139 } catch (const char* msg) {
140 std::cerr << "Error occurred during Matrix constructor: " << std::endl;
141 std::cerr << msg << (err ? W : H) << " Input Size: " << size << std::endl;
142 }
143}

534template<unsigned H, typename T>
535Mat<H-1, H-1, T> Mat<H, H, T>::SubMatrix(const unsigned int i, const unsigned int j) const {
536 Mat<H-1, H-1, T> Ms;
537 for(unsigned int i2 = 0, i3 = 0; i2 < H; i2++){
538 if(i2 == i) // skip row
539 continue;
540 for(unsigned int j2 = 0, j3 = 0; j2 < H; j2++){
541 if(j2 == j) // skip column
542 continue;
543 Ms[i3][j3] = (*this)[i2][j2];
544 j3++;
545 }
546 i3++;
547 }
548 return Ms;
549}

586template<unsigned H, typename T>
587double Mat<H, H, T>::Determinant(void) const {
588 // gaussian elimination
589 double det = 1;
590 double detDiv = 1;
591 // use a long double as big matrices can get large with this calculation methode
592 Mat<H, H, long double> GJR = (*this); // Matrix to execute the reduction on
593
594 unsigned int i = 0;
595 for(; i < H-1; i++){
596 // PN = PN * P0[0] - P0 * PN[0] => [0, 0, ...,] 0, a, b, c, ...
597 // divide determinant by GJR[i][i] for every loop
598 // det /= (H - (i + 1)) * GJR[i][i];
599 for(unsigned int j = i+1; j < H; j++){
600 detDiv *= GJR[i][i];
601 GJR[j] = (GJR[j] * GJR[i][i]) - (GJR[i] * GJR[j][i]);
602 }
603 }
604
605 for(unsigned int i = 0; i < H; i++){
606 det *= GJR[i][i];
607 }
608
609 det /= detDiv;
610
611 return det;
612}

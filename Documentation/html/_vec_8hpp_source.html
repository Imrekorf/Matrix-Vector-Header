525class Quaternion : public Vec<4, double> {
526public:
527 Quaternion() : Vec<4, double>(1, 0, 0, 0) {}
528 Quaternion(double w, double x, double y, double z) : Vec<4, double>(x, y, z, w) {}

535 const Quaternion getHProduct(const Quaternion q) const{
536 // Quaternion multiplication is defined by:
537 // (Q1 * Q2).w = (w1w2 - x1x2 - y1y2 - z1z2)
538 // (Q1 * Q2).x = (w1x2 + x1w2 + y1z2 - z1y2)
539 // (Q1 * Q2).y = (w1y2 - x1z2 + y1w2 + z1x2)
540 // (Q1 * Q2).z = (w1z2 + x1y2 - y1x2 + z1w2
541 return Quaternion(
542 w*q.w - x*q.x - y*q.y - z*q.z, // new w
543 w*q.x + x*q.w + y*q.z - z*q.y, // new x
544 w*q.y - x*q.z + y*q.w + z*q.x, // new y
545 w*q.z + x*q.y - y*q.x + z*q.w); // new z
546 }
551 const Quaternion getConjugate() const {
552 return Quaternion(w, -x, -y, -z);
553 }
554};

564template<int S, typename T>
565Vec<3, T>& Vec<S, T>::rotate(const Quaternion& q){
566 static_assert(S == 3, "Quaternion rotation needs a vector of size 3");
567 // http://www.cprogramming.com/tutorial/3d/quaternions.html
568 // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/transforms/index.htm
569 // http://content.gpwiki.org/index.php/OpenGL:Tutorials:Using_Quaternions_to_represent_rotation
570 // ^ or: http://webcache.googleusercontent.com/search?q=cache:xgJAp3bDNhQJ:content.gpwiki.org/index.php/OpenGL:Tutorials:Using_Quaternions_to_represent_rotation&hl=en&gl=us&strip=1
571
572 // P_out = q * P_in * conj(q)
573 // - P_out is the output vector
574 // - q is the orientation quaternion
575 // - P_in is the input vector (a*aReal)
576 // - conj(q) is the conjugate of the orientation quaternion (q=[w,x,y,z], q*=[w,-x,-y,-z])
577 Quaternion p(0, x, y, z);
578
579 // quaternion multiplication: q * p, stored back in p
580 p = q.getHProduct(p);
581
582 // quaternion multiplication: p * conj(q), stored back in p
583 p = p.getHProduct(q.getConjugate());
584
585 // p quaternion is now [0, x', y', z']
586 x = p.x;
587 y = p.y;
588 z = p.z;
589 return *this;
590}
